            </div>

        </form>

    <script>

        const rockMenu = [

            {

                drink: "Espresso Frappuccino",

                toppings: "Chips Frappuccino y Salsa Mocha",

            },

            {

                drink: "Cold Brew",

                toppings: "Cold Foam de Chocolate y Espiral de Mocha",

            },

            {

                drink: "Arequipe Latte",

                toppings: "Salsa de Dulce de Leche y Espiral de Caramelo",

            },

        ];


        const romanticMenu = [

            {

                drink: "Vainilla Latte Frío",

                toppings: "Cold Foam de Vainilla y Salsa de Caramelo",

            },

            {

                drink: "Strawberry Frappuccino",

                toppings: "Salsa de Fresa y Leche de Coco",

            },

            {

                drink: "Chocolate Blanco Caliente",

                toppings: "Espiral de Caramelo y Chips Frappuccino",

            },

        ];


        function generateMenu() {

            const genre = document.getElementById("genre").value;

            const song = document.getElementById("song").value;

            const resultDiv = document.getElementById("menuResult");

            

            let menu;

            if (genre === "rock") {

                menu = rockMenu;

            } else if (genre === "romantic") {

                menu = romanticMenu;

            }


            resultDiv.style.display = "block";

            resultDiv.innerHTML = `<h3>Your Menu Based on: ${song}</h3>`;

            menu.forEach(item => {

                resultDiv.innerHTML += `<p><strong>${item.drink}</strong> with ${item.toppings}</p>`;

            });

        }

    </script>
